## 最佳方案：工业级 Intent 层（增强版）

在原始方案基础上，整合评估建议，形成**高准确率、低成本、易维护**的意图识别架构。

### 一、整体架构图

```mermaid
graph TD
    U[用户输入] --> B[OpenClaw Bridge]
    B --> C{IntentCache}
    C -->|命中| D[返回缓存意图]
    C -->|未命中| E[L1 规则层]
    E -->|明确| F[记录缓存并返回]
    E -->|模糊| G[L2 向量记忆层]
    G -->|高置信度| F
    G -->|低置信度| H[L2.5 轻量级分类器]
    H -->|高置信度| F
    H -->|低置信度| I[L3 LLM 分类器]
    I --> J{置信度校准}
    J -->|≥动态阈值| K[意图切换检测]
    J -->|<阈值| L[澄清引擎]
    K -->|无切换| F
    K -->|检测切换| L
    L --> M[生成澄清问题]
    M --> N[更新Session状态]
    N --> O[返回用户]
```

```python
# apps/orchestra/rule_config.py
import yaml
import re
from pathlib import Path
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler
from typing import List, Dict, Optional

class RuleConfig:
    """支持热加载的规则配置"""
    
    def __init__(self, config_path: str = "config/intent_rules.yaml"):
        self.config_path = Path(config_path)
        self.casual_patterns: List[re.Pattern] = []
        self.marketing_patterns: List[re.Pattern] = []
        self.subtype_map: Dict[re.Pattern, str] = {}
        self.marketing_anchors: List[str] = []
        self._load_rules()
        self._start_watcher()
    
    def _load_rules(self):
        with open(self.config_path, 'r', encoding='utf-8') as f:
            config = yaml.safe_load(f)
        
        self.casual_patterns = [re.compile(p, re.IGNORECASE) for p in config.get('casual', [])]
        self.marketing_patterns = [re.compile(p, re.IGNORECASE) for p in config.get('marketing', [])]
        self.subtype_map = {re.compile(p, re.IGNORECASE): subtype for p, subtype in config.get('subtypes', {}).items()}
        self.marketing_anchors = config.get('marketing_anchors', [])
    
    def _start_watcher(self):
        """监听配置文件变化，自动热加载"""
        class Handler(FileSystemEventHandler):
            def __init__(self, rule_config):
                self.rule_config = rule_config
            def on_modified(self, event):
                if event.src_path == str(self.rule_config.config_path):
                    self.rule_config._load_rules()
                    print("规则配置已热加载")
        observer = Observer()
        observer.schedule(Handler(self), str(self.config_path.parent))
        observer.start()
```

```python
# apps/orchestra/memory_store.py
from typing import Optional, List
import numpy as np

class HybridMemoryStore:
    """
    混合记忆：用户级 + 全局级
    """
    def __init__(self, vector_store, global_threshold=0.8):
        self.vector_store = vector_store
        self.global_threshold = global_threshold
        # 预置全局热门意图模式（例如从运营数据中统计）
        self.global_hot_patterns = [
            ("账号诊断", "marketing", "diagnosis"),
            ("写文案", "marketing", "content_creation"),
            ("找选题", "marketing", "strategy"),
        ]
    
    async def similarity_search(self, text: str, user_id: str, k: int = 3) -> List:
        """先查用户级，若不足则补充全局级"""
        user_results = await self.vector_store.similarity_search(
            query=text, user_id=user_id, k=k, threshold=0.85
        )
        if len(user_results) >= k:
            return user_results
        
        # 补充全局匹配
        global_results = []
        for pattern, intent, subtype in self.global_hot_patterns:
            if pattern in text:
                global_results.append(MockQueryResult(intent=intent, subtype=subtype, confidence=0.75))
        # 也可用向量检索全局记忆库
        return user_results + global_results[:k - len(user_results)]
    
    async def get_prior_probability(self, user_id: str) -> float:
        """计算用户营销意图先验概率"""
        # 若无历史，根据用户来源设置（例如从 marketing_campaign 页面跳转）
        # 这里简化：若无历史则返回 0.3
        history_count = await self.vector_store.get_user_history_count(user_id)
        if history_count < 5:
            return 0.3
        # 有历史则计算营销意图占比
        ratio = await self.vector_store.get_marketing_ratio(user_id)
        return min(0.9, max(0.1, ratio))
```

```python
# apps/orchestra/lightweight_classifier.py
from transformers import AutoTokenizer, AutoModelForSequenceClassification
import torch
import asyncio

class LightweightClassifier:
    """微调 BERT 分类器，作为 L2.5 层"""
    def __init__(self, model_name="bert-base-chinese-finetuned-intent"):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForSequenceClassification.from_pretrained(model_name)
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.model.to(self.device)
        self.labels = ["casual", "marketing", "ambiguous"]
    
    async def predict(self, text: str) -> tuple[str, float]:
        loop = asyncio.get_event_loop()
        label, conf = await loop.run_in_executor(None, self._predict_sync, text)
        return label, conf
    
    def _predict_sync(self, text: str):
        inputs = self.tokenizer(text, return_tensors="pt", truncation=True, max_length=128).to(self.device)
        with torch.no_grad():
            logits = self.model(**inputs).logits
        probs = torch.softmax(logits, dim=1).cpu().numpy()[0]
        idx = int(probs.argmax())
        return self.labels[idx], float(probs[idx])
```

#### 2.4 置信度校准与动态阈值

```python
# apps/orchestra/confidence_calibrator.py
import pickle
from sklearn.isotonic import IsotonicRegression
import numpy as np

class ConfidenceCalibrator:
    """使用 Isotonic Regression 校准 LLM 置信度"""
    def __init__(self, model_path: str = "models/calibrator.pkl"):
        self.model_path = model_path
        self.calibrator = None
        self._load_or_create()
    
    def _load_or_create(self):
        try:
            with open(self.model_path, 'rb') as f:
                self.calibrator = pickle.load(f)
        except FileNotFoundError:
            # 初始化为恒等映射
            self.calibrator = IsotonicRegression(out_of_bounds='clip')
            self.calibrator.fit(np.array([0.0, 1.0]), np.array([0.0, 1.0]))
    
    def calibrate(self, raw_confidence: float) -> float:
        return float(self.calibrator.predict([[raw_confidence]])[0])
    
    def update(self, raw_confs: List[float], actual_accuracies: List[float]):
        """在线更新校准器（根据用户反馈）"""
        self.calibrator.fit(np.array(raw_confs), np.array(actual_accuracies))
        with open(self.model_path, 'wb') as f:
            pickle.dump(self.calibrator, f)

class DynamicThreshold:
    """基于用户活跃度和意图切换惩罚的动态阈值"""
    def __init__(self, base_threshold=0.7):
        self.base = base_threshold
    
    def get_threshold(self, user_id: str, session_context: dict) -> float:
        # 高活跃用户（历史对话>50）可降低阈值
        history_count = session_context.get("user_history_count", 0)
        if history_count > 50:
            threshold = self.base - 0.1
        elif history_count > 20:
            threshold = self.base - 0.05
        else:
            threshold = self.base
        
        # 如果上一轮意图是营销且用户没有切换意图的表示，略微降低阈值
        if session_context.get("previous_intent") == "marketing":
            threshold -= 0.05
        
        return max(0.5, min(0.85, threshold))
```

```python
# apps/orchestra/intent_switcher.py
class IntentSwitchDetector:
    def __init__(self, switch_penalty_threshold=2):
        self.switch_penalty_threshold = switch_penalty_threshold
    
    def detect(self, new_intent: Intent, session_context: dict) -> bool:
        """返回 True 表示检测到可能的话题切换，需要澄清"""
        last_intent = session_context.get("previous_intent")
        if not last_intent:
            return False
        
        if new_intent.type != last_intent:
            # 记录切换计数
            count = session_context.get("intent_switch_count", 0) + 1
            session_context["intent_switch_count"] = count
            if count >= self.switch_penalty_threshold:
                return True
        else:
            # 同意图则重置计数
            session_context["intent_switch_count"] = 0
        return False
```

#### 2.6 增强型 IntentEngine（整合所有改进）

```python
# apps/orchestra/intent_engine_enhanced.py
import hashlib
import time
from typing import Optional, Dict
from dataclasses import dataclass
from enum import Enum

class IntentType(Enum):
    CASUAL = "casual"
    MARKETING = "marketing"
    AMBIGUOUS = "ambiguous"
    SYSTEM = "system"

@dataclass
class Intent:
    type: IntentType
    subtype: Optional[str] = None
    confidence: float = 0.0
    entities: Dict = None
    reason: str = ""
    suggested_sop: Optional[str] = None

class IntentEngineEnhanced:
    def __init__(self, rule_config, memory_store, vector_store, llm_client,
                 calibrator=None, dynamic_threshold=None, switch_detector=None,
                 lightweight_classifier=None):
        self.rule_config = rule_config
        self.memory_store = memory_store
        self.vector_store = vector_store  # 用于实际存储
        self.llm_client = llm_client
        self.calibrator = calibrator or ConfidenceCalibrator()
        self.threshold_manager = dynamic_threshold or DynamicThreshold()
        self.switch_detector = switch_detector or IntentSwitchDetector()
        self.lightweight = lightweight_classifier  # 可选
        self.cache = IntentCache()
    
    async def recognize(self, text: str, user_id: str, session_context: dict) -> Intent:
        # 1. 缓存检查（考虑上下文哈希）
        cache_key = self._get_cache_key(text, user_id, session_context)
        cached = await self.cache.get(cache_key)
        if cached:
            return cached
        
        # 2. L1 规则层
        intent = self._layer1_rules(text)
        if intent.type != IntentType.AMBIGUOUS:
            await self.cache.set(cache_key, intent, ttl=300)
            return intent
        
        # 3. L2 向量记忆层（增强：混合记忆）
        intent = await self._layer2_memory(text, user_id)
        if intent.confidence > 0.85:
            await self.cache.set(cache_key, intent, ttl=300)
            return intent
        
        # 4. L2.5 轻量级分类器（如果启用）
        if self.lightweight:
            label, conf = await self.lightweight.predict(text)
            if conf > 0.9:
                intent = Intent(
                    type=IntentType(label),
                    confidence=conf,
                    reason="lightweight_classifier"
                )
                if intent.type != IntentType.AMBIGUOUS:
                    await self.cache.set(cache_key, intent, ttl=300)
                    return intent
        
        # 5. L3 LLM 分类器
        intent = await self._layer3_llm(text, session_context)
        
        # 6. 置信度校准
        calibrated_conf = self.calibrator.calibrate(intent.confidence)
        intent.confidence = calibrated_conf
        
        # 7. 动态阈值判定
        threshold = self.threshold_manager.get_threshold(user_id, session_context)
        if intent.confidence < threshold:
            # 转为模糊意图，由澄清引擎处理
            intent.type = IntentType.AMBIGUOUS
            intent.reason = f"low_confidence_after_calibration ({calibrated_conf:.2f} < {threshold:.2f})"
        
        # 8. 意图切换检测（若为营销/闲聊且可能切换）
        if intent.type != IntentType.AMBIGUOUS:
            if self.switch_detector.detect(intent, session_context):
                # 强制转为模糊，让澄清引擎确认是否切换
                intent.type = IntentType.AMBIGUOUS
                intent.reason = "potential_topic_switch"
        
        # 9. 缓存结果
        await self.cache.set(cache_key, intent, ttl=300)
        return intent
    
    def _layer1_rules(self, text: str) -> Intent:
        # 使用配置化的规则
        if any(p.search(text) for p in self.rule_config.casual_patterns):
            return Intent(type=IntentType.CASUAL, confidence=1.0, reason="hard_rule_casual")
        for pattern in self.rule_config.marketing_patterns:
            if pattern.search(text):
                subtype = self._extract_subtype(text)
                return Intent(type=IntentType.MARKETING, subtype=subtype, confidence=0.95, reason="hard_rule_marketing")
        return Intent(type=IntentType.AMBIGUOUS, confidence=0.0, reason="need_deeper_analysis")
    
    async def _layer2_memory(self, text: str, user_id: str) -> Intent:
        similar = await self.memory_store.similarity_search(text, user_id, k=3)
        if similar:
            marketing_count = sum(1 for q in similar if q.metadata.get("intent") == "marketing")
            if marketing_count >= 2:
                return Intent(
                    type=IntentType.MARKETING,
                    confidence=0.82,
                    reason="memory_similarity",
                    subtype=similar[0].metadata.get("subtype")
                )
        return Intent(type=IntentType.AMBIGUOUS, confidence=0.0, reason="no_memory_match")
    
    async def _layer3_llm(self, text: str, session_context: dict) -> Intent:
        # 构建带上下文的 Prompt（包含先验概率）
        prior = await self.memory_store.get_prior_probability(session_context.get("user_id", ""))
        prompt = self._build_classifier_prompt(text, session_context, prior)
        response = await self.llm_client.complete(prompt, response_format={"type": "json_object"}, temperature=0.1)
        # 使用 Pydantic 强校验
        validated = LLMIntentResponse.model_validate(response)
        return Intent(
            type=IntentType(validated.intent_type),
            subtype=validated.subtype,
            confidence=validated.confidence,
            reason="llm_classifier",
            entities=validated.entities,
            suggested_sop=validated.suggested_sop
        )
    
    def _get_cache_key(self, text: str, user_id: str, ctx: dict) -> str:
        # 缓存键包含上一轮意图和话题，避免上下文不同导致误命中
        last_intent = ctx.get("previous_intent", "")
        last_topic = ctx.get("previous_topic", "")
        raw = f"{user_id}:{text}:{last_intent}:{last_topic}"
        return hashlib.md5(raw.encode()).hexdigest()
```

```python
# apps/orchestra/clarification_enhanced.py
class ClarificationEngineEnhanced:
    def __init__(self, intent_engine):
        self.intent_engine = intent_engine
    
    async def handle_ambiguous(self, text: str, user_id: str, session: dict) -> dict:
        # 1. 优先尝试指代消解
        if self._is_coreference(text):
            resolved = await self._resolve_coreference(text, session)
            if resolved["resolved"]:
                return resolved
        
        # 2. 基于可能意图生成选项
        possible = self._guess_possible_intents(text, session)
        if len(possible) == 1:
            # 只有一个可能，直接确认
            return {
                "requires_clarification": True,
                "questions": [f"你是想{self._intent_to_chinese(possible[0])}吗？请确认一下。"],
                "suggestions": [self._intent_to_chinese(possible[0]), "不是"],
                "context": "single_possible"
            }
        elif len(possible) >= 2:
            options = [self._intent_to_chinese(p) for p in possible]
            return {
                "requires_clarification": True,
                "questions": [f"请问你是想：{'，还是'.join(options)}？"],
                "suggestions": options,
                "context": "multiple_possible"
            }
        else:
            # 完全未知，给出通用引导 + 基于热门意图的推荐
            hot_intents = await self._get_hot_intents()
            return {
                "requires_clarification": True,
                "questions": ["我没太理解你的问题。你是想咨询营销相关的问题，还是其他？"],
                "suggestions": hot_intents,
                "context": "complete_unknown"
            }
    
    async def _get_hot_intents(self) -> List[str]:
        # 从全局统计中获取最常见的三种意图
        return ["账号诊断", "文案生成", "选题建议"]
```

### 三、与 OpenClaw 的增强型 Bridge

```javascript
// packages/mcp-bridge/index.js
const axios = require('axios');
const { v4: uuidv4 } = require('uuid');

class IntentAwareBridge {
    constructor(pythonOrchestraUrl, timeoutMs = 3000) {
        this.pythonUrl = pythonOrchestraUrl;
        this.timeout = timeoutMs;
    }

    async execute(params, context) {
        const session = context.session;
        const sessionContext = {
            previous_intent: session.get('last_intent'),
            previous_topic: session.get('last_topic'),
            user_history_count: session.get('history_count', 0),
            intent_switch_count: session.get('intent_switch_count', 0),
            user_id: session.user_id,
        };

        // 异步调用意图识别（带超时和重试）
        let intentResult;
        try {
            intentResult = await this.callWithRetry({
                name: "intent_recognize",
                arguments: {
                    text: params.text,
                    user_id: session.user_id,
                    session_context: sessionContext
                }
            }, 2); // 最多重试2次
        } catch (err) {
            console.error('Intent recognition failed', err);
            // fallback: 假设为营销意图（保守）
            intentResult = { intent_type: 'marketing', confidence: 0.5, requires_clarification: false };
        }

        // 处理澄清状态
        if (intentResult.requires_clarification) {
            session.setState('CLARIFYING', {
                questions: intentResult.questions,
                possible_intents: intentResult.suggestions,
                original_text: params.text,
                timestamp: Date.now()
            });
            // 支持流式响应时，如果用户要求流式，需特殊处理
            if (params.streamMode) {
                return this.createClarificationStream(intentResult);
            }
            return {
                type: 'clarification',
                message: intentResult.questions[0],
                suggestions: intentResult.suggestions
            };
        }

        // 更新会话上下文
        session.set('last_intent', intentResult.intent_type);
        session.set('last_topic', intentResult.topic || params.text);
        session.set('intent_switch_count', 0); // 重置切换计数
        session.increment('history_count');

        // 调用实际业务逻辑（营销编排）
        const businessResult = await this.callWithRetry({
            name: "marketing_orchestra",
            arguments: {
                user_input: params.text,
                confirmed_intent: intentResult.intent_type,
                intent_subtype: intentResult.subtype,
                user_id: session.user_id
            }
        });
        return businessResult;
    }

    async callWithRetry(request, retries) {
        for (let i = 0; i <= retries; i++) {
            try {
                const response = await axios.post(`${this.pythonUrl}/call_tool`, request, {
                    timeout: this.timeout,
                    headers: { 'X-Request-Id': uuidv4() }
                });
                return response.data;
            } catch (err) {
                if (i === retries) throw err;
                await new Promise(r => setTimeout(r, 100 * (i + 1)));
            }
        }
    }

    createClarificationStream(intentResult) {
        // 实现流式返回澄清消息的简单示例
        const { Readable } = require('stream');
        const stream = new Readable({
            read() {
                this.push(intentResult.questions[0]);
                this.push(null);
            }
        });
        return { type: 'clarification_stream', stream };
    }
}
```

### 六、风险应对

1. **规则膨胀**：定期（每周）运行规则分析脚本，删除命中率低于 1% 的规则。
2. **LLM 幻觉**：使用 Pydantic 强校验 + 营销锚点二次过滤，拒绝不合规输出。
3. **高延迟**：L2.5 分类器推理 <10ms，且支持异步并行；设置 LLM 调用超时 2s，超时后降级为 L2 结果或澄清。
4. **数据隐私**：向量存储前对用户 ID 哈希处理，不存储明文敏感信息。

### 七、总结

本增强方案通过**规则配置化、混合记忆、轻量分类、置信度校准、动态阈值、切换检测**六大改进，在保持原有架构清晰度的前提下，显著提升了准确率和成本效率。可直接作为生产级 Intent 层的最佳实践进行落地。
